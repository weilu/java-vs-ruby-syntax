1 import java.util.*;
 2 
 3 class Programmer {
 4   private String name;
 5   private int age;
 6   private List<String> languages;
 7 
 8   public Programmer(String name) {
 9     this.name = name;
10     this.languages = new ArrayList<String>();
11   }
12 
13   public void learn(String languages) {
14     this.languages.add(languages);
15   }
16 
17   public String getName() {
18     return name;
19   }
20 
21   public void setAge(int age) {
22     this.age = age;
23   }
24 
25   public int getAge() {
26     return age;
27   }
28 
29   public String toString() {
30     return "A programmer named " + name
31       + ", " + age + " years old, knows " + languagesString() + ".";
32   }
33 
34   private String languagesString() {
35     String result = Arrays.toString(languages.toArray());
36     return result.substring(1, result.length() - 1);
37   }
38 
39   public static void main(String[] args) {
40     Programmer alice = new Programmer("Alice");
41     alice.setAge(8);
42     String[] languages = { "Java", "Ruby", "Python" };
43 
44     for(int i = 0; i < languages.length; i++) {
45       alice.learn(languages[i]);
46     }
47     System.out.println(alice);
48   }
49 }
